# 人才池表结构变动

## 人才池

```sql
alter table MY_HR_Reserve_TalentPool
    add create_user bigint
go

exec sp_addextendedproperty 'MS_Description', '创建人', 'SCHEMA', 'dbo', 'TABLE', 'MY_HR_Reserve_TalentPool', 'COLUMN',
     'create_user'
```
