# while 语句

```c++
#include <iostream>

using namespace std;

/**
 * while 语句
 */

int main() {
    // 三要素
    // 1. 初始条件
    int product = 2;
    while (product <= 1000) {   // 2. 循环条件
        product *= 2;   // 3. 条件改变
    }
}
```

# 铁球落地问题

```c++
#include <iostream>

using namespace std;

/**
 * 铁球落地问题
 * 物理问题：
 *      从飞机上落下来一个铁球，下落的高度 d(米) 和时间 t(秒) 之间的关系是 d = (1 / 2) g * t^2
 *      其中 g = 9.82m/s^2，假设飞机高度是 2500 米，需要多少秒落地？（精确到 1 秒）
 */

int main() {
    // 使用公式可以在 O(1) 时间内解决，为了体验 while 循环，此处用循环解决
    int t = 0;
    double d = 0;
    double g = 9.82;

    while (d < 2500) {
        t++;
        d = 0.5 * g * t * t;
    }

    cout << "需要 " << t << " 秒" << endl;
}
```
